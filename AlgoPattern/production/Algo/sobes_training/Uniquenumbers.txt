public class Uniquenumbers {
    public static void main (String[] args) {
	int[] inputArray = {1, 2, 3, 4, 2, 5, 3, 4, 6}
	
	// Объявляем HashSet для уникальных:
	HashSet<Integer> uniqueSet = new HashSet<>();
	// Объявляем List для дубликатов:
	List<Integer> duplicates = new ArrayList<>();
	
	for (int num : inputArray) {
	//Дубликат в лист
	    if(!uniqueSet.add(num)) {
		diplicates.add(num);
	    }

	}

	// HashSet в массив int[] uniqueArray
	int[] uniqueArray = uniqueSet.stream().mapToInt(Integer->intValue).toArray();
	int[] diplicatesArray = uniqueSet.stream().mapToInt(Integer->intValue).toArray();
	
	//Выводим Unique
	System.out.println("Уникальные:");
	for (int num : uniqueArray) {
	    System.out.print(num + " ");
	}

	//Выводим дубликаты	
	System.out.println("Дубликаты:");
	for (int num : diplicatesArray) {
	    System.out.print(num + " ");
	}
    }
}
